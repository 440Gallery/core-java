class Car{
	int price;
	String color;
	String version;
	String brand;
	
	Car(){
		this.price=price;
		this.color=color;
		this.version=version;
		this.brand=brand;
	}
	void show(){
		System.out.println(price);
		System.out.println(color);
		System.out.println(version);
		System.out.println(brand);
	}
}
